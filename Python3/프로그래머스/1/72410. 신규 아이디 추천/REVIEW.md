# 제목: 정규표현식을 활용한 O(N) 풀이
- 적용 조건: 문제의 요구사항이 명확한 문자열 치환/제거 규칙으로 이루어져 있을 때 효과적입니다.
- 핵심 아이디어: 문제의 각 단계를 `re.sub`, `strip` 등 명확한 의미의 내장 함수/메서드로 치환하여 가독성과 성능을 동시에 개선합니다.

import re

def solution_regex(new_id):
    # 1, 2단계: 소문자화 및 불필요한 문자 제거
    new_id = new_id.lower()
    new_id = re.sub(r'[^a-z0-9\-_.]', '', new_id)
    # 3단계: 연속된 마침표 하나로 치환
    new_id = re.sub(r'\.{2,}', '.', new_id)
    # 4단계: 처음/끝 마침표 제거
    new_id = new_id.strip('.')
    # 5단계: 빈 문자열이면 "a" 대입
    if not new_id:
        new_id = 'a'
    # 6단계: 길이 15자로 제한 및 끝 마침표 제거
    new_id = new_id[:15].rstrip('.')
    # 7단계: 길이가 2자 이하면 3자가 될 때까지 마지막 문자 반복
    while len(new_id) < 3:
        new_id += new_id[-1]
    return new_id
